// Use of all arithmetic operaters
#include<stdio.h>
int main() {

    float a,b,sum,difference,product,qutiont;
    printf("Enter First Number =");
    scanf("%f",&a);

    printf("Enter Second Number =");
    scanf("%f",&b);

    sum = a + b;
    printf("The sum of %f and %f is :%f \n",a,b,sum);

    difference = a - b;
    printf("The difference of %f and %f is :%f \n",a,b,difference);

    product = a * b;
    printf("The product of %f and %f is :%f\n",a,b,product);

    qutiont = a / b;
    printf("The qutiont of %f and %f is :%f\n",a,b,qutiont);

    return 0;
}
